  <script>
    // Данные
    let currentUser = {
      loggedIn: false,
      name: 'Пользователь',
      email: 'stan@example.com',
      about: 'О себе',
      role: 'admin',
      avatar: null
    };

    let news = [
      { title: 'Запуск нового сайта', content: 'Полностью переработанный дизайн с личным кабинетом и админ-панелью.', date: '17.03.2026' },
      { title: 'Встреча выпускников', content: 'Приглашаем на традиционную встречу 25 марта.', date: '15.03.2026' }
    ];

    // Навигация
    function showPage(pageId) {
      document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
      document.getElementById(pageId).classList.add('active');
      document.querySelectorAll('.nav-link').forEach(l => l.classList.remove('active'));
      
      if (pageId === 'pageNews') document.getElementById('navNews').classList.add('active');
      if (pageId === 'pageContacts') document.getElementById('navContacts').classList.add('active');
    }

    // Обновление UI авторизации
    function updateAuthUI() {
      const authBtn = document.getElementById('authButtons');
      const profileSec = document.getElementById('profileSection');
      
      if (currentUser.loggedIn) {
        authBtn.style.display = 'none';
        profileSec.style.display = 'flex';
        document.getElementById('profileAvatar').textContent = currentUser.name[0];
        updateAdminPanel();
        renderProfile();
      } else {
        authBtn.style.display = 'flex';
        profileSec.style.display = 'none';
      }
    }

    // Отрисовка профиля
    function renderProfile() {
      document.getElementById('profileName').value = currentUser.name;
      document.getElementById('profileEmail').value = currentUser.email;
      document.getElementById('profileAbout').value = currentUser.about;
      document.getElementById('userRole').value = currentUser.role;
    }

    // Админ-панель
    function updateAdminPanel() {
      const panel = document.getElementById('adminPanel');
      if (currentUser.loggedIn && currentUser.role === 'admin') {
        panel.classList.add('show');
      }
    }

    // Загрузка аватара
    document.getElementById('avatarInput').addEventListener('change', (e) => {
      const file = e.target.files[0];
      if (file) {
        const reader = new FileReader();
        reader.onload = (e) => {
          currentUser.avatar = e.target.result;
          const preview = document.getElementById('avatarPreview');
          preview.src = currentUser.avatar;
          preview.classList.add('has-image');
        };
        reader.readAsDataURL(file);
      }
    });

    // Форма профиля
    document.getElementById('profileForm').addEventListener('submit', (e) => {
      e.preventDefault();
      currentUser.name = document.getElementById('profileName').value;
      currentUser.email = document.getElementById('profileEmail').value;
      currentUser.about = document.getElementById('profileAbout').value;
      
      document.getElementById('profileAvatar').textContent = currentUser.name[0];
      alert('✅ Профиль обновлен!');
      updateAuthUI();
    });

    // Админ-форма
    document.getElementById('adminForm').addEventListener('submit', (e) => {
      e.preventDefault();
      const title = document.getElementById('postTitle').value;
      const content = document.getElementById('postContent').value;
      
      news.unshift({
        title,
        content,
        date: new Date().toLocaleDateString('ru-RU')
      });
      
      document.getElementById('adminForm').reset();
      renderNews();
      alert('✅ Новость опубликована!');
    });

    // Авторизация
    document.getElementById('loginBtn').onclick = () => {
      currentUser.loggedIn = true;
      updateAuthUI();
      alert('✅ Добро пожаловать!');
    };

    document.getElementById('registerBtn').onclick = () => {
      currentUser.loggedIn = true;
      currentUser.role = 'user';
      updateAuthUI();
      alert('✅ Регистрация завершена!');
    };

    // Навигация
    document.getElementById('logoHome').onclick = () => showPage('pageHome');
    document.getElementById('navNews').onclick = (e) => { e.preventDefault(); showPage('pageNews'); renderNews(); };
    document.getElementById('navContacts').onclick = (e) => { e.preventDefault(); showPage('pageContacts'); };
    document.getElementById('profileBtn').onclick = (e) => { e.preventDefault(); showPage('pageProfile'); };

    // Отрисовка новостей
    function renderNews() {
      const container = document.getElementById('newsList');
      container.innerHTML = news.map(item => `
        <div class="news-item card">
          <h3>${item.title}</h3>
          <div class="news-date">${item.date}</div>
          <p>${item.content}</p>
        </div>
      `).join('');
    }

    // Инициализация
    renderNews();
  </script>
